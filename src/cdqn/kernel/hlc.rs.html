3use std::sync::{Arc, Mutex};
4use std::time::{SystemTime, UNIX_EPOCH};
5
6// The HLC service. It holds the last known time and a counter.
7#[derive(Debug, Clone)]
8pub struct HLC {
9    // We now store the time as microseconds since the Unix epoch.
10    last_time_micros: Arc<Mutex<u128>>,
11    counter: Arc<Mutex<u16>>,
12}
13
14impl Default for HLC {
15    fn default() -> Self {
16        Self::new()
17    }
18}
19
20impl HLC {
21    pub fn new() -> Self {
22        let now_micros = SystemTime::now()
23            .duration_since(UNIX_EPOCH)
24            .expect("Time went backwards")
25            .as_micros();
26        HLC {
27            last_time_micros: Arc::new(Mutex::new(now_micros)),
28            counter: Arc::new(Mutex::new(0)),
29        }
30    }
31
32    // Generates a new, unique, and sortable timestamp ID.
33    pub fn now(&self) -> (String, String) {
34        let mut last_time = self.last_time_micros.lock().unwrap();
35        let mut counter = self.counter.lock().unwrap();
36
37        let current_time = SystemTime::now()
38            .duration_since(UNIX_EPOCH)
39            .expect("Time went backwards")
40            .as_micros();
41
42        if current_time > *last_time {
43            *last_time = current_time;
44            *counter = 0;
45        } else {
46            *counter += 1;
47        }
48
49        // We no longer have a convenient RFC3339 formatter, so we'll just use the micros.
50        // This is a temporary simplification.
51        let timestamp_utc = format!("{}", *last_time);
52        let kdu_id = format!("{}-{:04x}", *last_time, *counter);
53
54        (kdu_id, timestamp_utc)
55    }
56}
