3use crate::kernel::KDU;
4use std::collections::HashMap;
5
6/// The EntityScheduler is the heart of the runtime's event loop.
7/// It is responsible for delivering KDUs to entities and executing their behaviors.
8pub struct EntityScheduler {
9    // We will need a way to store and look up our entities and their mailboxes.
10    // A HashMap from an entity's FQEI (String) to its Mailbox is a good start.
11    mailboxes: HashMap<String, super::entity::Mailbox>,
12}
13
14impl EntityScheduler {
15    /// Creates a new, empty scheduler.
16    pub fn new() -> Self {
17        EntityScheduler {
18            mailboxes: HashMap::new(),
19        }
20    }
21
22    /// The main event loop. This function will run forever, processing messages.
23    pub fn run(&mut self) {
24        println!("[EntityScheduler] Starting main event loop...");
25        loop {
26            // In the next milestone, we will implement the logic here to:
27            // 1. Check all mailboxes for new messages.
28            // 2. Execute the entity's behavior for one message.
29            // 3. Route the resulting KDUs to their destination mailboxes.
30
31            // For now, we just prevent an infinite busy-loop.
32            std::thread::sleep(std::time::Duration::from_secs(1));
33        }
34    }
35}
36
37// Implement the Default trait as suggested by clippy.
38impl Default for EntityScheduler {
39    fn default() -> Self {
40        Self::new()
41    }
