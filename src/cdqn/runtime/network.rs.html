3use crate::kernel::KDU;
4use std::io::{self, Read};
5use std::net::{TcpListener, TcpStream};
6use std::sync::mpsc::Sender;
7use std::thread::{self, JoinHandle};
8
9pub struct NodeServer;
10
11impl NodeServer {
12    /// Spawns the network server in a new thread.
13    /// Returns a JoinHandle for the thread.
14    pub fn spawn(addr: &str, kdu_tx: Sender<KDU>) -> JoinHandle<()> {
15        let listener = TcpListener::bind(addr).expect("Failed to bind to network address");
16        println!("[NodeServer] Thread started. Listening on {}", addr);
17
18        thread::spawn(move || {
19            for stream in listener.incoming() {
20                match stream {
21                    Ok(stream) => {
22                        println!("[NodeServer] New connection: {}", stream.peer_addr().unwrap());
23                        let kdu_tx_clone = kdu_tx.clone();
24                        thread::spawn(move || {
25                            handle_client(stream, kdu_tx_clone);
26                        });
27                    }
28                    Err(e) => {
29                        eprintln!("[NodeServer] Connection failed: {}", e);
30                    }
31                }
32            }
33        })
34    }
35}
36
37/// The client handler now sends the received KDU back to the Orchestrator.
38fn handle_client(mut stream: TcpStream, kdu_tx: Sender<KDU>) {
39    let mut len_buffer = [0u8; 8];
40    if stream.read_exact(&mut len_buffer).is_ok() {
41        let kdu_len = u64::from_le_bytes(len_buffer);
42        let mut kdu_buffer = vec![0; kdu_len as usize];
43
44        if stream.read_exact(&mut kdu_buffer).is_ok() {
45            let kdu: KDU =
46                bincode::deserialize(&kdu_buffer).expect("Failed to deserialize KDU from stream");
47            println!("[NodeServer] Received KDU with ID: {}. Sending to Orchestrator.", kdu.kdu_id);
48            kdu_tx.send(kdu).expect("Failed to send KDU to orchestrator");
49        }
50    }
51}

53// NodeClient remains unchanged for now.
54pub struct NodeClient;
55impl NodeClient {
56    pub fn connect(addr: &str) -> io::Result<TcpStream> {
57        TcpStream::connect(addr)
58    }
59    pub fn send_kdu(stream: &mut TcpStream, kdu: &KDU) -> io::Result<()> {
60        let kdu_bytes = bincode::serialize(kdu).expect("Failed to serialize KDU");
61        let kdu_len = kdu_bytes.len() as u64;
62        stream.write_all(&kdu_len.to_le_bytes())?;
63        stream.write_all(&kdu_bytes)?;
64        Ok(())
65    }
